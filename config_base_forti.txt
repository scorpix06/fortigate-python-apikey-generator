# Script d'automatisation de la configuration de base des Fortigate
# v7.2.4


# Config de base

config system global
set admin-sport 4443
set timezone 26
set admin-https-redirect disable
end

config system settings
set central-nat enable
end

# Config radius

config user radius
edit "RTSRADIUS"
set server "rtsradius.riviera-telecom.com"
set secret Monaco.2017
next
config accounting-server
edit 1
set status enable
set server "rtsradius.riviera-telecom.com"
set secret Monaco.2017
set port 1813
next
end

config system admin
edit "RTSRADIUS"
set remote-auth enable
set accprofile "super_admin"
set wildcard enable
set remote-group "RTSRADIUS"
end



# Config VPN

config vpn ssl setting
set default-portal "tunnel-access"
set source-interface wan1
set source-interface wan
set port 10443
edit 1 
set groups "RTSRADIUS"
set portal "tunnel-access"
next
end
